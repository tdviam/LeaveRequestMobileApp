CREATE TABLE DELEGATION_MANAGEMENT (
	DELEGATION_ID	INTEGER	NOT NULL	GENERATED ALWAYS
	AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	OWNER	VARCHAR(50),
	DELEGATE	VARCHAR(50),
	START_DATE	TIMESTAMP,
	END_DATE	TIMESTAMP,
	STATUS	VARCHAR(20)
);

CREATE TABLE LEAVE_REQUEST (
	ID	INTEGER	NOT NULL	GENERATED ALWAYS
	AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	STAFF_CODE	VARCHAR(100)	NOT NULL,
	STAFF_NAME	VARCHAR(100)	NOT NULL,
	STAFF_USERNAME	VARCHAR(100)	NOT NULL,
	START_DATE	DATE	NOT NULL,
	END_DATE	DATE	NOT NULL,
	LEAVE_TYPE_ID	INTEGER,
	A1_NAME	VARCHAR(100),
	A1_STATUS	VARCHAR(100),
	A1_STATUS_SUB_DATE	DATE,
	A2_NAME	VARCHAR(100),
	A2_STATUS	VARCHAR(100),
	A2_STATUS_SUB_DATE	DATE,
	A3_NAME	VARCHAR(100),
	A3_STATUS	VARCHAR(100),
	A3_STATUS_SUB_DATE	DATE,
	A4_NAME	VARCHAR(100),
	A4_STATUS	VARCHAR(100),
	A4_STATUS_SUB_DATE	DATE,
	A5_NAME	VARCHAR(100),
	A5_STATUS	VARCHAR(100),
	A5_STATUS_SUB_DATE	DATE,
	FINAL_STATUS	VARCHAR(100),
	BALANCE	INTEGER	DEFAULT ,
	REQUEST_DETAILS	VARCHAR(500),
	INSTANCE_ID	VARCHAR(10)
);

CREATE TABLE LEAVE_TYPE (
	ID	INTEGER	NOT NULL	GENERATED ALWAYS
	AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	LEAVE_NAME	VARCHAR(100)	NOT NULL,
	LEAVE_DESCRIPTION	VARCHAR(100)
);

CREATE TABLE POLICY (
	ID	INTEGER	NOT NULL	GENERATED ALWAYS
	AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	POLICY_NAME	VARCHAR(100)	NOT NULL
);

CREATE TABLE POLICY_CONFIGURATION (
	ID	INTEGER	NOT NULL	GENERATED ALWAYS
	AS IDENTITY (START WITH 1, INCREMENT BY 1, CACHE 20,
	   NO MINVALUE, NO MAXVALUE, NO CYCLE, NO ORDER),
	POLICY_ID	INTEGER	NOT NULL,
	BALANCE	INTEGER	NOT NULL,
	LEAVE_TYPE_ID	INTEGER	NOT NULL,

CREATE TABLE POLICY_USER_ASSIGMENT (
